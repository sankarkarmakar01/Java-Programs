import java.io.*;

import java.util.*;

public class Solution {
	public static ArrayList<Integer> getAllDivisors(int n){
		// Write your code here.
		ArrayList<Integer> ans = new ArrayList<>();
		int i = 1;
		while(i <= n) {
			if(n % i == 0){
				ans.add(i);
			}
			i++;
		}
		return ans;
	}
}
