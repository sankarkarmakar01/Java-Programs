import java.util.* ;
import java.io.*;
public class Solution
{
	public static boolean isArmstrong(int num)
	{
	    // wrtie your code here;
		int count = 0;
		int originalNumber = num;
		int n = num;
		int rev = 0;
		while(num != 0) {
			int rem = num % 10;
			count++;
			num /= 10;
		}
		while(n != 0) {
			int rem = n % 10;
			rev = rev + (int)(Math.pow(rem,count));
			n /= 10;
		}
		if(originalNumber == rev) {
			return true;
		}
		return false;
	}
}
