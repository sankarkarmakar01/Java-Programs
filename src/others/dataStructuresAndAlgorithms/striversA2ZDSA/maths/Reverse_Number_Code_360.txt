import java.util.* ;
import java.io.*;

public class Solution {
	public static long reverseNumber(long n) {
		// Write your code here
		long rev = 0L;
		while(n != 0) {
			long rem = n % 10;
			rev = (rev * 10) + rem;
			n /= 10;
		}
		return rev;
	}

}
